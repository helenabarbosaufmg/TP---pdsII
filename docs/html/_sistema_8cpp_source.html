277void Sistema::criarReserva(Reserva r) {
278 int codigoHospede = r.RHospede().HCodigo();
279 if (!hospedeCadastrado(codigoHospede)) {
280 cout << "Erro: hospede nao esta cadastrado." << endl;
281 return;
282 }
283
284 int numeroQuarto = r.RQuarto().QNumero();
285 int posQuarto = posicaoQuarto(numeroQuarto);
286 if (posQuarto == -1) {

288 return;
289 }
290
291 string checkIn = r.RCheckIn();
292 string checkOut = r.RCheckOut();

295 cout << "Erro: formato de data invalido (use dd/mm/aaaa)." << endl;

297 }
298
299

302 return;
303 }
304
305 reservas.push_back(r);
306 reservas.back().setStatus("Agendada");
